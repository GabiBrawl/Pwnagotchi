# RTC Sync Service - Syncs hardware clock to system clock on boot
[Unit]
Description=Sync RTC Clock on Boot
DefaultDependencies=no
Before=sysinit.target

[Service]
Type=oneshot
# Read time from hardware clock and set system time
ExecStart=/sbin/hwclock --hctosys

[Install]
WantedBy=sysinit.target
